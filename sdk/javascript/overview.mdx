Build powerful chat experiences in your web applications using CometChat's JavaScript SDK. Whether you're building with React, Angular, Vue, Next.js, or vanilla JavaScript, this SDK provides the foundation for real-time messaging, voice, and video calling.

<Note>
**Choose Your Integration Path**

| Approach | Best For | Effort | Customization |
|----------|----------|--------|---------------|
| [Chat Widget](/widget/html/overview) | Quick embed, any website | No Code | Low |
| [UI Kit Builder](/chat-builder/react/overview) | Visual customization | Low Code | Medium |
| [UI Kits](/ui-kit/react/overview) | Pre-built components | Code | High |
| **SDK** (this guide) | Complete flexibility | Code | Full |
| [REST API](https://api-explorer.cometchat.com) | Server-side operations | Code | Full |

**Not sure?** Start with [UI Kits](/ui-kit/react/overview) for faster development, or use the SDK if you need complete control over the UI.
</Note>

## Quick Start

Get chat working in under 5 minutes:

<Steps>
  <Step title="Get Your Credentials">
    1. [Sign up for CometChat](https://app.cometchat.com/signup) (free trial available)
    2. Create a new app and select your region
    3. Go to **API & Auth Keys** and copy:
       - **App ID**
       - **Region** (`us`, `eu`, or `in`)
       - **Auth Key**
  </Step>
  
  <Step title="Install the SDK">
    <Tabs>
      <Tab title="npm">
        ```bash
        npm install @cometchat/chat-sdk-javascript
        ```
      </Tab>
      <Tab title="yarn">
        ```bash
        yarn add @cometchat/chat-sdk-javascript
        ```
      </Tab>
      <Tab title="pnpm">
        ```bash
        pnpm add @cometchat/chat-sdk-javascript
        ```
      </Tab>
      <Tab title="CDN">
        ```html
        <script src="https://unpkg.com/@cometchat/chat-sdk-javascript/CometChat.js"></script>
        ```
      </Tab>
    </Tabs>
  </Step>
  
  <Step title="Initialize & Login">
    ```javascript
    import { CometChat } from "@cometchat/chat-sdk-javascript";

    // Replace with your credentials
    const APP_ID = "YOUR_APP_ID";
    const REGION = "YOUR_REGION";
    const AUTH_KEY = "YOUR_AUTH_KEY";

    // Initialize CometChat
    const appSettings = new CometChat.AppSettingsBuilder()
      .subscribePresenceForAllUsers()
      .setRegion(REGION)
      .autoEstablishSocketConnection(true)
      .build();

    await CometChat.init(APP_ID, appSettings);
    console.log("✓ CometChat initialized");

    // Login with a test user
    const user = await CometChat.login("cometchat-uid-1", AUTH_KEY);
    console.log("✓ Logged in as:", user.getName());
    ```
    
    <Info>
    **Test Users:** We provide 5 pre-created users for testing: `cometchat-uid-1` through `cometchat-uid-5`
    </Info>
  </Step>
  
  <Step title="Send Your First Message">
    ```javascript
    // Send a message to another test user
    const message = new CometChat.TextMessage(
      "cometchat-uid-2",              // Receiver UID
      "Hello from CometChat! 👋",     // Message text
      CometChat.RECEIVER_TYPE.USER    // Receiver type
    );

    const sentMessage = await CometChat.sendMessage(message);
    console.log("✓ Message sent:", sentMessage.getId());
    ```
  </Step>
  
  <Step title="Receive Messages">
    ```javascript
    // Listen for incoming messages
    CometChat.addMessageListener(
      "MY_LISTENER",
      new CometChat.MessageListener({
        onTextMessageReceived: (message) => {
          console.log("New message from:", message.getSender().getName());
          console.log("Text:", message.getText());
        }
      })
    );
    ```
  </Step>
</Steps>

<Check>
**That's it!** You now have a working chat implementation. Open two browser tabs, login as different test users, and send messages between them.
</Check>

## Complete Example

Here's a full working example you can copy and run:

<Tabs>
  <Tab title="Complete Script">
    ```javascript
    import { CometChat } from "@cometchat/chat-sdk-javascript";

    // ============================================
    // CONFIGURATION - Replace with your credentials
    // ============================================
    const APP_ID = "YOUR_APP_ID";
    const REGION = "YOUR_REGION";
    const AUTH_KEY = "YOUR_AUTH_KEY";

    // ============================================
    // INITIALIZE
    // ============================================
    async function initCometChat() {
      const appSettings = new CometChat.AppSettingsBuilder()
        .subscribePresenceForAllUsers()
        .setRegion(REGION)
        .autoEstablishSocketConnection(true)
        .build();

      try {
        await CometChat.init(APP_ID, appSettings);
        console.log("✓ CometChat initialized");
        return true;
      } catch (error) {
        console.error("✗ Initialization failed:", error);
        return false;
      }
    }

    // ============================================
    // LOGIN
    // ============================================
    async function login(uid) {
      try {
        // Check if already logged in
        let user = await CometChat.getLoggedinUser();
        if (user) {
          console.log("✓ Already logged in as:", user.getName());
          return user;
        }

        // Login
        user = await CometChat.login(uid, AUTH_KEY);
        console.log("✓ Logged in as:", user.getName());
        return user;
      } catch (error) {
        console.error("✗ Login failed:", error);
        return null;
      }
    }

    // ============================================
    // SEND MESSAGE
    // ============================================
    async function sendMessage(receiverUID, text) {
      const message = new CometChat.TextMessage(
        receiverUID,
        text,
        CometChat.RECEIVER_TYPE.USER
      );

      try {
        const sent = await CometChat.sendMessage(message);
        console.log("✓ Message sent:", sent.getText());
        return sent;
      } catch (error) {
        console.error("✗ Send failed:", error);
        return null;
      }
    }

    // ============================================
    // RECEIVE MESSAGES
    // ============================================
    function setupMessageListener(onMessageReceived) {
      CometChat.addMessageListener(
        "MESSAGE_LISTENER",
        new CometChat.MessageListener({
          onTextMessageReceived: (message) => {
            console.log("📩 New message from:", message.getSender().getName());
            console.log("   Text:", message.getText());
            onMessageReceived?.(message);
          },
          onMediaMessageReceived: (message) => {
            console.log("📎 Media received:", message.getAttachment()?.getUrl());
            onMessageReceived?.(message);
          }
        })
      );
    }

    // ============================================
    // FETCH MESSAGE HISTORY
    // ============================================
    async function getMessages(uid, limit = 30) {
      const messagesRequest = new CometChat.MessagesRequestBuilder()
        .setUID(uid)
        .setLimit(limit)
        .build();

      try {
        const messages = await messagesRequest.fetchPrevious();
        console.log(`✓ Fetched ${messages.length} messages`);
        return messages;
      } catch (error) {
        console.error("✗ Fetch failed:", error);
        return [];
      }
    }

    // ============================================
    // RUN
    // ============================================
    async function main() {
      // 1. Initialize
      const initialized = await initCometChat();
      if (!initialized) return;

      // 2. Login
      const user = await login("cometchat-uid-1");
      if (!user) return;

      // 3. Setup listener
      setupMessageListener((message) => {
        // Handle new messages here
      });

      // 4. Fetch history with another user
      const messages = await getMessages("cometchat-uid-2");

      // 5. Send a message
      await sendMessage("cometchat-uid-2", "Hello! 👋");

      console.log("\n🎉 CometChat is ready!");
    }

    main();
    ```
  </Tab>
  <Tab title="React Hook">
    ```jsx
    // hooks/useCometChat.js
    import { useEffect, useState, useCallback } from "react";
    import { CometChat } from "@cometchat/chat-sdk-javascript";

    const APP_ID = "YOUR_APP_ID";
    const REGION = "YOUR_REGION";
    const AUTH_KEY = "YOUR_AUTH_KEY";

    export function useCometChat() {
      const [isReady, setIsReady] = useState(false);
      const [user, setUser] = useState(null);
      const [error, setError] = useState(null);

      // Initialize on mount
      useEffect(() => {
        const init = async () => {
          try {
            const appSettings = new CometChat.AppSettingsBuilder()
              .subscribePresenceForAllUsers()
              .setRegion(REGION)
              .autoEstablishSocketConnection(true)
              .build();

            await CometChat.init(APP_ID, appSettings);
            
            // Check for existing session
            const loggedInUser = await CometChat.getLoggedinUser();
            if (loggedInUser) {
              setUser(loggedInUser);
            }
            
            setIsReady(true);
          } catch (err) {
            setError(err.message);
          }
        };

        init();
      }, []);

      // Login function
      const login = useCallback(async (uid) => {
        try {
          const loggedInUser = await CometChat.login(uid, AUTH_KEY);
          setUser(loggedInUser);
          return loggedInUser;
        } catch (err) {
          setError(err.message);
          throw err;
        }
      }, []);

      // Logout function
      const logout = useCallback(async () => {
        await CometChat.logout();
        setUser(null);
      }, []);

      // Send message function
      const sendMessage = useCallback(async (receiverUID, text) => {
        const message = new CometChat.TextMessage(
          receiverUID,
          text,
          CometChat.RECEIVER_TYPE.USER
        );
        return CometChat.sendMessage(message);
      }, []);

      return {
        isReady,
        user,
        error,
        login,
        logout,
        sendMessage,
        CometChat
      };
    }

    // Usage in component:
    // const { isReady, user, login, sendMessage } = useCometChat();
    ```
  </Tab>
</Tabs>
