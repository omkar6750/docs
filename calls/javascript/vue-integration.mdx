This guide walks you through integrating the CometChat Calls SDK into a Vue application. By the end, you'll have a working video call implementation with proper lifecycle handling.

## Prerequisites

Before you begin, make sure you've completed the [prerequisites](/calls/javascript/overview#prerequisites) from the Overview page and ensure you have a working Vue project.

## Step 2: Initialize and Login

### Initialize

Initialize the SDK with your app credentials. Call this once when your application starts.

```javascript
import { CometChatCalls } from "@cometchat/calls-sdk-javascript";

const { error } = await CometChatCalls.init({
  appId: "APP_ID", // Replace with your App ID
  region: "REGION", // Replace with your Region (us, eu, in)
  authKey: "AUTH_KEY", // (Optional) Replace with your Auth Key
});

if (error) {
  console.error("CometChatCalls initialization failed:", error);
} else {
  console.log("CometChatCalls initialized successfully");
}
```

### Login

Log in the user with their UID. Ensure the SDK is initialized before calling `login()`.

```javascript
CometChatCalls.login("UID") // Replace with your user's UID
  .then((user) => {
    console.log("Logged in successfully", user);
  });
```

## Step 3: Generate Token

Generate a call token for the session. The call token is required to authenticate the user for that particular session.

```javascript
CometChatCalls.generateToken("SESSION_ID") // Replace with your Session ID
  .then(({ token }) => {
    console.log("token", token);
  });
```

## Step 4: Join Session

Use the generated token to join the session along with your call settings and a container element.

```typescript
const sessionContainer = ref<HTMLDivElement | null>(null);

const sessionSettings: SessionSettings = {};
CometChatCalls.joinSession(token, sessionSettings, sessionContainer.value!);

// In your template
// <div ref="sessionContainer" style="width: 100vw; height: 100vh" />
```

For all available call settings options, see [Session Settings](/calls/javascript/session-settings).

## Step 5: Minimum Working Example

Here's a complete Vue component that initializes the SDK, logs in, generates a token, and joins a call session:

```vue App.vue
<script setup lang="ts">
import { ref, onUnmounted } from "vue";
import {
  CometChatCalls,
  type SessionSettings,
} from "@cometchat/calls-sdk-javascript";

const sessionContainerRef = ref<HTMLDivElement | null>(null);
const inCall = ref(false);

async function startCall() {
  // 1. Initialize
  const { error } = await CometChatCalls.init({
    appId: "APP_ID",
    region: "REGION",
    authKey: "AUTH_KEY",
  });

  if (error) {
    console.error("CometChatCalls initialization failed:", error);
    return;
  }

  // 2. Login
  await CometChatCalls.login("cometchat-uid-1");

  // 3. Generate Token
  const { token } = await CometChatCalls.generateToken("demo");

  // 4. Join Session — ref is always mounted so it's safe to use directly
  const sessionSettings: SessionSettings = {};
  inCall.value = true;
  CometChatCalls.joinSession(
    token,
    sessionSettings,
    sessionContainerRef.value!,
  );
}

const unsub = CometChatCalls.addEventListener("onConnectionClosed", () => {
  inCall.value = false;
});

onUnmounted(() => {
  if (inCall.value) {
    CometChatCalls.leaveSession();
  }
  unsub();
});
</script>

<template>
  <div
    :style="{
      display: 'flex',
      justifyContent: 'center',
      alignItems: 'center',
      width: '100vw',
      height: '100vh',
    }"
  >
    <button
      @click="startCall"
      :style="{
        padding: '12px 32px',
        fontSize: '18px',
        cursor: 'pointer',
        borderRadius: '8px',
        border: 'none',
        backgroundColor: '#6851D6',
        color: '#fff',
      }"
    >
      Start Call
    </button>
  </div>

  <div
    ref="sessionContainerRef"
    :style="{
      position: 'fixed',
      top: 0,
      left: 0,
      width: '100vw',
      height: '100vh',
      zIndex: 9999,
      backgroundColor: '#000',
      visibility: inCall ? 'visible' : 'hidden',
    }"
  />
</template>
```
